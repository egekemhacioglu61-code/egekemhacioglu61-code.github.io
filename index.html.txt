<script>
const quotes = {
  basari: [
    "Başarı vazgeçmeyenlerindir.",
    "Disiplin özgürlüktür.",
    "Zor günler güçlü insanlar yaratır."
  ],
  spor: [
    "Ter başarının anahtarıdır.",
    "Bugün zor, yarın güçlü.",
    "Pes etmek yok."
  ],
  ask: [
    "Sevgi en büyük güçtür.",
    "Kalbinle yaşa.",
    "Aşk cesaret ister."
  ]
};

let currentCategory = "basari";

function randomQuote() {
  const list = quotes[currentCategory];
  const random = Math.floor(Math.random() * list.length);
  return list[random];
}

function newQuote() {
  const quoteElement = document.getElementById("quote");
  quoteElement.style.opacity = 0;

  setTimeout(() => {
    quoteElement.innerText = randomQuote();
    quoteElement.style.opacity = 1;
  }, 200);
}

function changeCategory(category) {
  currentCategory = category;
  newQuote();
}

function shareQuote() {
  const text = document.getElementById("quote").innerText;
  navigator.clipboard.writeText(text + " - EgoMotivasyon");
  alert("Söz kopyalandı! İstediğin yerde paylaşabilirsin.");
}

window.onload = function() {
  newQuote();
};
</script>
